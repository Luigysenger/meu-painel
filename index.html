    <script>

        const player = document.getElementById('audio-player');

        const grid = document.getElementById('stations-grid');

        const status = document.getElementById('status-display');

        

        function generateLightColor() {

            const r = Math.floor(Math.random() * 56) + 200;

            const g = Math.floor(Math.random() * 56) + 200;

            const b = Math.floor(Math.random() * 56) + 200;

            return `rgb(${r}, ${g}, ${b})`;

        }


        let radioList = JSON.parse(localStorage.getItem('luigy_radios_v2')) || [];

        let currentIndex = -1;

        let editingIndex = -1;


        // --- SISTEMA DE CLIMA E DATA ---

        function updateDateTime() {

            const now = new Date();

            document.getElementById('display-time').innerText = now.toLocaleTimeString('pt-BR');

            document.getElementById('display-date').innerText = now.toLocaleDateString('pt-BR', {day:'2-digit', month:'2-digit', year:'2-digit'});

        }

        setInterval(updateDateTime, 1000);


        async function getClima() {

            try {

                // Busca localização pelo IP

                const locRes = await fetch('https://ipapi.co/json/');

                const locData = await locRes.json();

                const city = locData.city;

                document.getElementById('display-city').innerText = city.toUpperCase();


                // Busca clima (Open-Meteo é grátis e não precisa de chave)

                const weatherRes = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${locData.latitude}&longitude=${locData.longitude}&current_weather=true&minutely_15=precipitation`);

                const weatherData = await weatherRes.json();

                

                const temp = Math.round(weatherData.current_weather.temperature);

                const code = weatherData.current_weather.weathercode; // Código do tempo

                const rain = weatherData.minutely_15 ? weatherData.minutely_15.precipitation[0] : 0;


                document.getElementById('display-temp').innerText = temp;


                // Lógica dos Ícones conforme seu pedido

                let icon = "☀️";

                if (rain > 0) {

                    icon = "🌧️"; // Chuva em 15-30 min

                } else if (code >= 1 && code <= 3) {

                    icon = "⛅"; // Nublado

                } else if (code > 3) {

                    icon = "☁️"; // Muito nublado

                }

                document.getElementById('display-icon').innerText = icon;


            } catch (e) {

                document.getElementById('display-city').innerText = "ERRO CLIMA";

            }

        }

        // Atualiza clima a cada 15 minutos

        getClima();

        setInterval(getClima, 900000);


        // --- LÓGICA DAS RÁDIOS (MANTIDA CONFORME COMBINADO) ---

        new Sortable(grid, {

            animation: 150,

            ghostClass: 'sortable-ghost',

            onEnd: function() {

                const cards = grid.querySelectorAll('.radio-card');

                const newList = [];

                cards.forEach(card => {

                    const id = card.getAttribute('data-id');

                    const radio = radioList.find(r => r.id == id);

                    newList.push(radio);

                });

                radioList = newList;

                saveToStorage();

            }

        });


        function checkEnter(event) { if (event.key === "Enter") handleLogin(); }

        function saveToStorage() { localStorage.setItem('luigy_radios_v2', JSON.stringify(radioList)); }


        function handleLogin() {

            const user = document.getElementById('user-input').value.toUpperCase();

            const pass = document.getElementById('pass-input').value;

            if (user === "LUIGY" && pass === "1045") {

                document.getElementById('login-screen').style.display = 'none';

                document.getElementById('main-panel').style.display = 'flex';

                getClima(); // Atualiza ao entrar

                renderGrid();

            } else { alert("Acesso Negado!"); }

        }


        function logout() {

            stopRadio();

            document.getElementById('main-panel').style.display = 'none';

            document.getElementById('login-screen').style.display = 'block';

        }


        function nextStep() {

            if(!document.getElementById('new-name').value) return alert("Digite o nome!");

            document.getElementById('step-1').classList.add('hidden');

            document.getElementById('step-2').classList.remove('hidden');

        }


        function addNewRadio() {

            let name = document.getElementById('new-name').value;

            let url = document.getElementById('new-url').value;

            if(editingIndex > -1) {

                radioList[editingIndex] = { ...radioList[editingIndex], name, url };

                editingIndex = -1;

            } else {

                radioList.push({ id: Date.now(), name, url, color: generateLightColor() });

            }

            saveToStorage();

            renderGrid();

            resetForm();

        }


        function resetForm() {

            document.getElementById('new-name').value = "";

            document.getElementById('new-url').value = "";

            document.getElementById('step-2').classList.add('hidden');

            document.getElementById('step-1').classList.remove('hidden');

        }


        function renderGrid() {

            grid.innerHTML = "";

            radioList.forEach((radio, index) => {

                const card = document.createElement('div');

                card.className = `radio-card ${currentIndex === index ? 'active' : ''}`;

                card.setAttribute('data-id', radio.id);

                card.style.backgroundColor = radio.color;

                card.innerHTML = `

                    <div class="card-actions">

                        <button class="btn-card" onclick="event.stopPropagation(); editRadio(${index})">✏️</button>

                        <button class="btn-card" style="color:red" onclick="event.stopPropagation(); removeRadio(${index})">×</button>

                    </div>

                    <div onclick="playByIndex(${index})">

                        <div style="font-size:22px; margin-bottom:5px">${currentIndex === index ? '🔊' : '▶️'}</div>

                        <div style="font-size:13px; font-weight:bold;">${radio.name}</div>

                    </div>

                `;

                grid.appendChild(card);

            });

        }


        function playByIndex(index) {

            currentIndex = index;

            renderGrid();

            const radio = radioList[currentIndex];

            status.innerText = "Tocando: " + radio.name;

            player.src = radio.url;

            player.play().catch(() => status.innerText = "Erro no sinal!");

        }


        function stopRadio() {

            player.pause();

            currentIndex = -1;

            renderGrid();

            status.innerText = "Parado.";

        }


        function nextRadio() {

            if (radioList.length === 0) return;

            let next = (currentIndex + 1) >= radioList.length ? 0 : currentIndex + 1;

            playByIndex(next);

        }


        function prevRadio() {

            if (radioList.length === 0) return;

            let prev = (currentIndex - 1) < 0 ? radioList.length - 1 : currentIndex - 1;

            playByIndex(prev);

        }


        function removeRadio(index) {

            if(confirm("Remover?")) {

                radioList.splice(index, 1);

                saveToStorage();

                renderGrid();

            }

        }
