// --- CLIMA REFEITO (ESTRATÉGIA DE SEGURANÇA) --- async function getClima() { const cidadeTxt = document.getElementById('cidade-info'); const tempTxt = document.getElementById('temp-atual'); const iconDiv = document.getElementById('clima-icon'); try { // 1. Pega a localização pelo IP (Sem pedir permissão ao usuário) const locRes = await fetch('https://ipwho.is/'); const locData = await locRes.json(); if(!locData.success) throw new Error("Falha na localização"); const cidade = locData.city; const lat = locData.latitude; const lon = locData.longitude; // 2. Busca o clima usando as coordenadas exatas do IP const weatherRes = await fetch(`https://api.openweathermap.org/data/2.5/weather?lat=${lat}&lon=${lon}&units=metric&appid=895284fb2d2c1d877c724822b7485303&lang=pt_br`); const wData = await weatherRes.json(); // 3. Atualiza a tela cidadeTxt.innerText = cidade.toUpperCase(); tempTxt.innerText = Math.round(wData.main.temp) + "°C"; // Puxa o ícone oficial (Sol, Chuva, Nuvens, etc) const iconCode = wData.weather[0].icon; iconDiv.innerHTML = ``; console.log("Clima carregado com sucesso para:", cidade); } catch (error) { console.error("Erro no clima:", error); cidadeTxt.innerText = "SÃO PAULO (PADRÃO)"; // Busca clima de SP se a localização falhar const fallback = await fetch(`https://api.openweathermap.org/data/2.5/weather?q=Sao%20Paulo&units=metric&appid=895284fb2d2c1d877c724822b7485303&lang=pt_br`); const fData = await fallback.json(); tempTxt.innerText = Math.round(fData.main.temp) + "°C"; iconDiv.innerHTML = ``; } } // Chama a função logo ao abrir getClima();
