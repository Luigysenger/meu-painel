// --- AJUSTE NO MONITOR DE OUVINTES (CORRIGIDO) ---         function toggleUserDetails(id) {             const bar = document.getElementById('details-' + id);             const isOpen = (bar.style.display === 'block');             bar.style.display = isOpen ? 'none' : 'block';             openUsers[id] = !isOpen;          }         function formatarTempoReal(ms) { // Se o tempo for menor que zero ou maior que 24 horas (ex: erro de timestamp 1970) // ele retorna zerado para evitar números absurdos na tela             if (!ms || ms < 0 || ms > 86400000) return "00:00:00";             const totalSegundos = Math.floor(ms / 1000);             const horas = Math.floor(totalSegundos / 3600);             const minutos = Math.floor((totalSegundos % 3600) / 60);             const segundos = totalSegundos % 60;             return `${horas.toString().padStart(2, '0')}:${minutos.toString().padStart(2, '0')}:${segundos.toString().padStart(2, '0')}`;         }         // Loop de atualização visual (Front-end)         setInterval(() => {             const spans = document.querySelectorAll('.tempo-live');             spans.forEach(span => {                 const startTime = parseInt(span.dataset.start); const agora = Date.now(); // Só calcula se o startTime for recente (nas últimas 24h)                 if (startTime > (agora - 86400000)) {                     span.innerText = formatarTempoReal(agora - startTime);                 } else { span.innerText = "00:00:00"; }             });         }, 1000);         db.ref('usuarios').on('value', snap => {             const container = document.getElementById('user-list');             container.innerHTML = "";             snap.forEach(c => {                 const u = c.val(), id = c.key; const agora = Date.now();                 const isOnline = (agora - parseInt(u.lastSeen || 0) < 60000);                 const dataOuvindo = u.lastSeen ? new Date(u.lastSeen).toLocaleString('pt-BR') : '---';                 let rádioAtual = u.radioAtual || 'Navegando...';                 let startTime = parseInt(u.radioStartTime || 0);                                  // Validação: Se o startTime for 0 ou maior que o tempo atual, resetamos a exibição let diff = agora - startTime; if (startTime === 0 || diff < 0 || diff > 86400000) { diff = 0; }                 let tempoHTML = isOnline                      ? `${formatarTempoReal(diff)}`                     : `${formatarTempoReal(u.tempoTotalRadio || 0)}`;                 const displayStyle = openUsers[id] ? 'block' : 'none';                 container.innerHTML += `                    
                       
                                                   
                       
                           

Rádio Atual: ${rádioAtual}

                           

Tempo nesta Rádio: ${tempoHTML}

                           

Última Atividade: ${dataOuvindo}

                       
                   
`;             });         });
