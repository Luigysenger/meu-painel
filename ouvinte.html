// --- AJUSTE NA PREVISÃO DO TEMPO POR IP (SILENCIOSO E REAL) --- async function getWeather() { try { // 1. Pega a localização pelo IP (sem pedir permissão de GPS) const locRes = await fetch('https://demo.ip-api.com/json/?fields=status,message,city,lat,lon'); const locData = await locRes.json(); if (locData.status !== 'success') throw new Error('Falha na localização'); const lat = locData.lat; const lon = locData.lon; const cidade = locData.city; // 2. Busca o clima real usando as coordenadas do IP const climaRes = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}¤t_weather=true`); const climaData = await climaRes.json(); if (climaData.current_weather) { updateWeatherUI(climaData.current_weather.temperature, climaData.current_weather.weathercode, cidade); } } catch (e) { console.error("Erro clima:", e); document.getElementById('city').innerText = "CLIMA INDISPONÍVEL"; } } function updateWeatherUI(temp, code, cidade) { document.getElementById('temp').innerText = Math.round(temp) + "°C"; document.getElementById('city').innerText = cidade.toUpperCase(); const iconEl = document.getElementById('weather-icon'); iconEl.className = "fas"; // Reset // Mapeamento preciso de códigos (WMO) para ícones // 0: Limpo, 1-3: Nublado, 45-48: Nevoeiro, 51-67: Chuva/Garoa, 71-77: Neve, 80-82: Pancadas, 95-99: Tempestade if (code === 0) { iconEl.classList.add("fa-sun"); } else if (code >= 1 && code <= 3) { iconEl.classList.add("fa-cloud-sun"); } else if (code >= 51 && code <= 67) { iconEl.classList.add("fa-cloud-showers-heavy"); // ÍCONE DE CHUVA } else if (code >= 80 && code <= 82) { iconEl.classList.add("fa-cloud-rain"); // ÍCONE DE PANCADAS } else if (code >= 95) { iconEl.classList.add("fa-bolt"); // TEMPESTADE } else if (code >= 45 && code <= 48) { iconEl.classList.add("fa-smog"); } else { iconEl.classList.add("fa-cloud"); // PADRÃO NUVEM } } getWeather(); setInterval(getWeather, 900000); // Atualiza a cada 15 minutos
